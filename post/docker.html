mongodb

docker run \
--name mymongo \
-d \
-e MONGO_INITDB_ROOT_USERNAME=root \
-e MONGO_INITDB_ROOT_PASSWORD=123456 \
-e MONGO_INITDB_DATABASE=admin \
-p 27017:27017 \
-v ~/container/mongo:/data/db \
mongo \
--auth
